function badgeClass(action) {

  return action.toLowerCase().replace(/ /g, '-');

}


function renderCard(item) {

  const changeClass = item.change_1m_pct >= 0 ? 'up' : 'down';

  const changeSign  = item.change_1m_pct >= 0 ? '+' : '';

  const upside = item.analyst_target !== 'N/A' && item.current_price

    ? (((item.analyst_target - item.current_price) / item.current_price) * 100).toFixed(1)

    : 'N/A';


  const newsHtml = (item.news || []).slice(0, 2)

    .map(h => `<div class="news-item">${h}</div>`).join('');


  const id = 'chart-' + item.ticker;


  return `

    <div class="card">

      <div class="card-header">

        <div>

          <div class="ticker">${item.ticker}</div>

          <div class="price">$${item.current_price}</div>

          <div class="change ${changeClass}">${changeSign}${item.change_1m_pct}% this month</div>

        </div>

        <div>

          <div class="badge ${badgeClass(item.action)}">${item.action}</div>

          <div style="text-align:right; margin-top:8px; font-size:22px; font-weight:800">${item.score}<span style="font-size:13px;color:#666">/10</span></div>

        </div>

      </div>


      <div class="score-bar"><div class="score-fill" style="width:${item.score * 10}%"></div></div>


      <div class="stats">

        <div class="stat">

          <div class="stat-label">Price Target</div>

          <div class="stat-value">$${item.price_target}</div>

        </div>

        <div class="stat">

          <div class="stat-label">Analyst Upside</div>

          <div class="stat-value">${upside !== 'N/A' ? upside + '%' : 'N/A'}</div>

        </div>

        <div class="stat">

          <div class="stat-label">P/E Ratio</div>

          <div class="stat-value">${typeof item.pe_ratio === 'number' ? item.pe_ratio.toFixed(1) : item.pe_ratio}</div>

        </div>

        <div class="stat">

          <div class="stat-label">Beta</div>

          <div class="stat-value">${typeof item.beta === 'number' ? item.beta.toFixed(2) : item.beta}</div>

        </div>

      </div>


      <canvas id="${id}" height="80"></canvas>


      <p class="summary">${item.summary}</p>

      <div class="risk">⚠️ ${item.key_risk}</div>

      ${newsHtml ? `<div style="margin-top:12px">${newsHtml}</div>` : ''}

    </div>

  `;

}


function drawChart(item) {

  const ctx = document.getElementById('chart-' + item.ticker);

  if (!ctx || !item.prices) return;

  const up = item.change_1m_pct >= 0;

  new Chart(ctx, {

    type: 'line',

    data: {

      labels: item.dates,

      datasets: [{

        data: item.prices,

        borderColor: up ? '#00e676' : '#ff5252',

        backgroundColor: up ? '#00e67611' : '#ff525211',

        borderWidth: 2,

        pointRadius: 0,

        fill: true,

        tension: 0.4

      }]

    },

    options: {

      plugins: { legend: { display: false } },

      scales: {

        x: { display: false },

        y: { display: false }

      }

    }

  });

}


async function loadData() {

  document.getElementById('loading').style.display = 'block';

  document.getElementById('main').style.display = 'none';


  const res  = await fetch('/api/data');

  const data = await res.json();


  // Top pick = highest score

  const top = data[0];

  document.getElementById('top-pick').innerHTML = `

    <h2>🏆 Top Pick</h2>

    <h3>${top.ticker} — $${top.current_price} <span class="badge ${badgeClass(top.action)}" style="font-size:14px">${top.action}</span></h3>

    <p>${top.summary}</p>

  `;


  document.getElementById('grid').innerHTML = data.map(renderCard).join('');


  document.getElementById('loading').style.display = 'none';

  document.getElementById('main').style.display = 'block';


  data.forEach(drawChart);

}


loadData();
